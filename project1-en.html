ERP & EC Process Optimization

Pain Point: Data silos and inefficient manual data transfer caused delays during peak order periods.

Sub ProcessData()
  Dim ws As Worksheet
  Dim lastRow As Long
  Dim cell As Range
  Dim dateValue As String
  Dim orderDict As Collection

  ' Set Worksheet (Assume Sheet 1)
  Set ws = ThisWorkbook.Sheets(1)
  ' Find Last Row
  lastRow = ws.Cells(ws.Rows.Count, "F").End(xlUp).Row

  ' Insert Columns & Add Headers
  ws.Columns("A:B").Insert Shift:=xlToRight
  ws.Cells(1, 1).Value = "Merge OrderID & Recipient"

  ' Format Date yyyymmdd
  For Each cell In ws.Range("F2:F" & lastRow)
    If IsDate(cell.Value) Then
      dateValue = Format(cell.Value, "yyyymmdd")
      cell.Value = dateValue
    End If
  Next cell

  ' Data Cleaning: Remove Special Chars
  For Each col In ws.UsedRange
    If Not IsEmpty(col.Value) Then
      col.Value = Replace(col.Value, "'", "")
    End If
  Next col

  ' Logic: Calculate Discounted Unit Price
  For Each cell In ws.Range("AC2:AC" & lastRow)
    If IsNumeric(cell.Value) Then
      salesAmount = cell.Value
      quantity = cell.Offset(0, -10).Value
      If quantity <> 0 Then
        cell.Offset(0, 1).Value = salesAmount / quantity
      End If
    End If
  Next cell

  ws.Columns("AC:AD").AutoFit
End Sub
