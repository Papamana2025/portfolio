Sub AutoProcess_ERP_Data()
  ' --------------------------------
  ' STEP 1: Initialize System
  ' --------------------------------
  Dim ws As Worksheet
  Dim lastRow As Long
  Set ws = ThisWorkbook.Sheets("RawData")

  ' 自動偵測最後一行數據
  lastRow = ws.Cells(ws.Rows.Count, "A").End(xlUp).Row
  Debug.Print "Found " & lastRow & " records."

  ' --------------------------------
  ' STEP 2: Data Cleaning Loop
  ' --------------------------------
  For Each cell In ws.Range("F2:F" & lastRow)
    If IsDate(cell.Value) Then
      ' 格式標準化 yyyymmdd
      cell.Value = Format(cell.Value, "yyyymmdd")
    End If
    ' 合併主單號與收件人資訊
    cell.Offset(0, 2).Value = CleanString(cell.Value)
  Next cell

  ' --------------------------------
  ' STEP 3: Export to ERP Format
  ' --------------------------------
  ws.Columns("A:B").Insert Shift:=xlToRight
  MsgBox "Process Complete. Ready for Upload."
End Sub
