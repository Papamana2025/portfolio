ERP & EC 營運流程再造

痛點：部門數據脫節，人工轉檔低效導致訂單高峰期延遲。

解方：不只是寫程式，更是「流程重塑」。利用 VBA 半自動化腳本，將多平台數據流無縫串接，釋放團隊生產力。

Sub ProcessData()
  Dim ws As Worksheet
  Dim lastRow As Long
  Dim cell As Range
  Dim dateValue As String

  ' 設定工作表
  Set ws = ThisWorkbook.Sheets(1)
  lastRow = ws.Cells(ws.Rows.Count, "F").End(xlUp).Row

  ' 格式清洗：轉換日期格式
  For Each cell In ws.Range("F2:F" & lastRow)
    If IsDate(cell.Value) Then
      dateValue = Format(cell.Value, "yyyymmdd")
      cell.Value = dateValue
    End If
  Next cell

  ' 數據標準化：移除特殊符號
  For Each col In ws.UsedRange
    If Not IsEmpty(col.Value) Then
      col.Value = Replace(col.Value, "'", "")
    End If
  Next col

  ' 邏輯運算：建立訂單序號字典
  Set orderDict = New Collection
  On Error Resume Next
  For Each cell In ws.Range("D2:D" & lastRow)
    orderID = cell.Value
    If IsEmpty(orderDict(orderID)) Then
      orderCounter = orderCounter + 1
      orderDict.Add orderCounter, orderID
    End If
    cell.Offset(0, -2).Value = orderDict(orderID)
  Next cell

  ' 商業邏輯：計算折扣後單價
  For Each cell In ws.Range("AC2:AC" & lastRow)
    If IsNumeric(cell.Value) Then
      salesAmount = cell.Value
      quantity = cell.Offset(0, -10).Value
      If quantity <> 0 Then
        cell.Offset(0, 1).Value = salesAmount / quantity
      End If
    End If
  Next cell

  ws.Columns("AC:AD").AutoFit
End Sub
